### Disabled

The radio group can be disabled using the isDisabled prop:

<Story
  name="disabled"
  parameters={{
    docs: {
      source: {
        code: `
<RadioGroup defaultValue="one" isDisabled>
  <Radio value="one">One</Radio>
  <Radio value="two">Two</Radio>
  <Radio value="three">Three</Radio>
</RadioGroup>
        `,
      },
    },
  }}
>
  <Box sx={{ pb: 3 }}>
    <RadioGroup defaultValue="one" isDisabled>
      <Radio value="one">One</Radio>
      <Radio value="two">Two</Radio>
      <Radio value="three">Three</Radio>
    </RadioGroup>
  </Box>
</Story>
