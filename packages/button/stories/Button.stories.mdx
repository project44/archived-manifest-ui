      Default
    </Button>
  </Flex>
</Story>

### Icons

The button can be decorated with icons using the `startIcon` and `endIcon` props:

<Story
  name="icons"
  parameters={{
    docs: {
      source: {
        code: `
<Button startIcon={<Add />}>Default</Button>
<Button endIcon={<ExpandMore />}>Default</Button>
<Button endIcon={<ExpandMore />} startIcon={<Add />}>Default</Button>
        `,
      },
    },
  }}
>
  <Flex align="center" gap={2} sx={{ pb: 3 }}>
    <Button startIcon={<Add />}>Default</Button>
    <Button endIcon={<ExpandMore />}>Default</Button>
    <Button startIcon={<Add />} endIcon={<ExpandMore />}>
      Default
    </Button>
  </Flex>
</Story>

### Danger

A button can be styled to signify destructive actions using the `colorTheme` prop:

<Story
  name="danger"
  parameters={{ docs: { source: { code: '<Button colorTheme="danger">Default</Button>' } } }}
>
  <Box sx={{ pb: 3 }}>
    <Button colorTheme="danger">Danger</Button>
  </Box>
</Story>

### Disabled

The button can be disabled using the `isDisabled` prop:

<Story
  name="disabled"
  parameters={{
    docs: {
      source: {
        code: `
<Button isDisabled variant="filled">Disabled</Button>

<Button colorTheme="neutral" isDisabled variant="text">Disabled</Button>
        `,
      },
    },
  }}
>
  <Flex align="center" gap={2} sx={{ pb: 3 }}>
    <Button isDisabled>Disabled</Button>
    <Button colorTheme="neutral" isDisabled variant="outlined">
      Disabled
    </Button>
    <Button colorTheme="neutral" isDisabled variant="text">
      Disabled
    </Button>
    <Button colorTheme="danger" isDisabled>
      Disabled
    </Button>
  </Flex>
</Story>
